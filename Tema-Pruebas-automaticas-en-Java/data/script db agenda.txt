CREATE SCHEMA AGENDA AUTHORIZATION DBA

CREATE TABLE TiposContacto (
  IdTipoContacto INTEGER GENERATED BY DEFAULT AS IDENTITY,
  TipoContacto VARCHAR(50) NULL,
  PRIMARY KEY (IdTipoContacto)
)

CREATE TABLE Contactos (
  IdContacto INTEGER GENERATED BY DEFAULT AS IDENTITY,
  Nombre VARCHAR(50) NULL,
  Apellidos VARCHAR(50) NULL,
  Estimado VARCHAR(50) NULL,
  Direccion VARCHAR(255) NULL,
  Ciudad VARCHAR(50) NULL,
  EdoOProv VARCHAR(20) NULL,
  CodPostal VARCHAR(20) NULL,
  Region VARCHAR(50) NULL,
  Pais VARCHAR(50) NULL,
  NombreCompania VARCHAR(50) NULL,
  Cargo VARCHAR(50) NULL,
  TelefonoTrabajo VARCHAR(30) NULL,
  ExtensionTrabajo VARCHAR(20) NULL,
  TelefonoMovil VARCHAR(30) NULL,
  NumFax VARCHAR(30) NULL,
  NomCorreoElectronico VARCHAR(50) NULL,
  FechaUltimaReunion DATETIME NULL,
  IdTipoContacto INTEGER NULL,
  ReferidoPor VARCHAR(50) NULL,
  Notas LONGVARCHAR NULL,
  PRIMARY KEY (IdContacto),
  FOREIGN KEY (IdTipoContacto)
    REFERENCES TiposContacto(IdTipoContacto)
)

CREATE TABLE Llamadas (
  IdLlamada INTEGER GENERATED BY DEFAULT AS IDENTITY,
  IdContacto INTEGER NULL,
  FechaLlamada DATE NULL,
  HoraLlamada TIME NULL,
  Asunto VARCHAR(255) NULL,
  Notas LONGVARCHAR NULL,  
  PRIMARY KEY (IdLlamada),
  FOREIGN KEY (IdContacto)
    REFERENCES Contactos(IdContacto)
)
